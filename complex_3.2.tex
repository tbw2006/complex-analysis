\section{引言}

在上一章（Cauchy 定理）中我们已经知道：如果 \(f\) 在闭合曲线 \(\gamma\) 内部及边界上解析，则
\[
\int_{\gamma} f(z)\,dz = 0.
\]
那么，如果 \(f\) 在 \(\gamma\) 内部有孤立奇点（例如极点），积分还会是零吗？答案是否定的。最简单的例子是 \(f(z)=1/z\)，绕原点一周的积分为 \(2\pi i\)。本节的核心问题正是：如何计算这种围道积分？答案就是著名的\textbf{留数公式}。

\section{主要结果：留数定理}

\subsection{圆盘情形}
\begin{theorem}[留数定理，圆盘情形]
设 \(f\) 在包含圆盘 \(\overline{D}\) 的某开集上解析，除去圆心处的一个极点 \(z_0\)，则
\[
\int_{C} f(z)\,dz = 2\pi i \cdot \operatorname{Res}_{z_0}f,
\]
其中 \(C\) 是正向圆周，\(\operatorname{Res}_{z_0}f\) 表示 \(f\) 在 \(z_0\) 处的留数。
\end{theorem}

\subsection{多个极点}
\begin{theorem}[留数定理，多个极点]
若 \(f\) 在圆盘内除有限个极点 \(z_1,\dots,z_N\) 外解析，则
\[
\int_{C} f(z)\,dz = 2\pi i \sum_{k=1}^{N} \operatorname{Res}_{z_k}f.
\]
\end{theorem}

\subsection{一般围道}
\begin{theorem}[留数定理，一般围道]
对任意\textbf{玩具围道}（toy contour）\(\gamma\) 及其内部，只要 \(f\) 在 \(\gamma\) 内部除有限个极点外解析，则有
\[
\int_{\gamma} f(z)\,dz = 2\pi i \sum_{\text{内部极点}} \operatorname{Res}_{z_k}f.
\]
\end{theorem}

\begin{remark}
这里“玩具围道”是指边界简单、内部可以明确定义的一类曲线（如圆、矩形、钥匙孔形等），其上 Cauchy 定理和留数公式均成立。
\end{remark}

\section{证明的逻辑脉络}

\subsection{简化到单个极点}
使用\textbf{钥匙孔围道}（keyhole contour）技巧：
\begin{itemize}
\item 构造一个大围道，包含所有极点，再在每个极点处挖一个很小的圆 \(C_{\epsilon}\) 绕开极点。
\item 让钥匙孔两侧的缝隙趋于零，由于 \(f\) 在缝隙两侧连续，积分相互抵消。
\item 最终得到：大围道上的积分 = 各个小圆上积分的和（注意小圆的取向与大围道相反）。
\end{itemize}

\subsection{计算单个小圆上的积分}
在极点 \(z_0\) 附近，\(f\) 可以展开为 \textbf{Laurent 级数}：
\[
f(z) = \frac{a_{-n}}{(z-z_0)^n} + \cdots + \frac{a_{-1}}{z-z_0} + G(z),
\]
其中 \(G(z)\) 解析（即不含负幂项）。
\begin{itemize}
\item 对 \(\frac{a_{-1}}{z-z_0}\) 绕小圆积分：
\[
\int_{C_{\epsilon}} \frac{a_{-1}}{z-z_0}\,dz = 2\pi i a_{-1}.
\]
\item 对 \(\frac{a_{-k}}{(z-z_0)^k}\;(k\ge 2)\) 积分：由于原函数为 \(\frac{a_{-k}}{(1-k)(z-z_0)^{k-1}}\) 是单值函数，闭路积分为 \(0\)。
\item 对 \(G(z)\)（解析）积分：由 Cauchy 定理为 \(0\)。
\end{itemize}
所以小圆积分 \(= 2\pi i a_{-1}\)，即 \(2\pi i\) 乘以留数。

\subsection{得到留数公式}
将所有极点贡献相加，即得
\[
\int_{\gamma} f(z)\,dz = 2\pi i \sum \operatorname{Res}.
\]

\section{留数的计算方法}
\begin{definition}
\(f\) 在孤立奇点 \(z_0\) 处的\textbf{留数}为 Laurent 展开中 \((z-z_0)^{-1}\) 项的系数，记为 \(\operatorname{Res}_{z_0}f\)。
\end{definition}

\begin{theorem}[留数的计算]
\begin{itemize}
\item 若 \(z_0\) 是\textbf{可去奇点}，则留数为 \(0\)。
\item 若 \(z_0\) 是\textbf{简单极点}（一阶极点）：
\[
\operatorname{Res}_{z_0}f = \lim_{z\to z_0} (z-z_0)f(z).
\]
\item 若 \(z_0\) 是 \(n\) 阶极点：
\[
\operatorname{Res}_{z_0}f = \frac{1}{(n-1)!}\lim_{z\to z_0}\frac{d^{n-1}}{dz^{n-1}}\big[(z-z_0)^n f(z)\big].
\]
\end{itemize}
\end{theorem}

\section{典型例子}

\subsection{例1：\(\displaystyle \int_{-\infty}^{\infty} \frac{dx}{1+x^2} = \pi\)}
考虑 \(f(z)=\frac{1}{1+z^2}\)，在 \(z=i\) 处有一阶极点（上半平面）。取上半平面的大半圆围道 \(\gamma_R\)（实轴线段 \([-R,R]\) + 上半圆弧 \(C_R^+\)）。由留数定理：
\[
\int_{\gamma_R} f(z)\,dz = 2\pi i \cdot \operatorname{Res}_{z=i}f = 2\pi i \cdot \frac{1}{2i} = \pi.
\]
当 \(R\to\infty\) 时，圆弧上的积分趋于 \(0\)（因为 \(|f(z)|\sim 1/R^2\)），所以实轴上的积分趋于 \(\pi\)。被积函数为偶函数，故 \(\int_{-\infty}^{\infty} = \pi\)。

\subsection{例2：\(\displaystyle \int_{-\infty}^{\infty} \frac{e^{ax}}{1+e^x}dx = \frac{\pi}{\sin\pi a},\quad 0<a<1\)}
用矩形围道（高 \(2\pi\)），顶点在 \(R,\, R+2\pi i,\,-R+2\pi i,\,-R\)。内部只有 \(z=\pi i\) 一个简单极点，留数为 \(-e^{a\pi i}\)。上下水平边的积分差给出
\[
(1-e^{2\pi i a})I = -2\pi i e^{a\pi i},
\]
解得 \(I = \pi/\sin\pi a\)。

\subsection{例3：\(\displaystyle \int_{-\infty}^{\infty} \frac{e^{-2\pi i x\xi}}{\cosh\pi x}dx = \frac{1}{\cosh\pi\xi}\)}
类似地用矩形围道（高 \(2\)），内部有极点 \(z=i/2\) 和 \(z=3i/2\)。通过留数计算和上下边积分抵消，得到傅里叶变换的自对偶性。

这些例子展示了留数公式的强大：将实积分转化为围道积分，再通过极点留数计算。

\section{逻辑总结}
整个第 3.2 节的逻辑可以概括为：
\begin{itemize}
\item \textbf{已知}：若 \(f\) 在闭曲线内部解析，积分为零（Cauchy 定理）；若 \(f\) 有极点，则 Laurent 展开中含有负幂次项，其中 \(-1\) 次项贡献非零积分。
\item \textbf{推导}：通过“挖掉极点”的技巧，将原围道积分化为围绕各个极点的小圆积分之和。每个小圆积分只由留数决定，高阶负幂项积分为零。因此得到\textbf{留数公式}。
\item \textbf{应用}：计算实积分（尤其是有理函数、三角函数、指数函数的 Fourier 变换），为后续的幅角原理、零点计数等打下基础。
\end{itemize}
